---
title: 阻止默认事件和事件传播
date: 2016-06-25
tags: [DOM,Summary,Function]
categories: Dynamic
---

```javascript
    var link = document.querySelector("a");
    link.addEventListener("click",function(e){
        e.preventDefault();
        console.log("link click");
        e.stopPropagation();
    },false);
    document.body.addEventListener("click",function(){
        console.log("body click");
    },false);
```
