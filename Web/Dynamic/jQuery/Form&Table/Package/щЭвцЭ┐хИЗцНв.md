```html
<style>
    html,body,div,ul,li{
        padding: 0;
        margin: 0;
    }
    .tab{
        width: 300px;
        height: 240px;
        margin: 20px auto;
        border: solid 1px;
        overflow: hidden;
    }
    ul{
        overflow: hidden;
    }
    li{
        float: left;
        border: solid 1px;
        height: 20px;
        list-style-type: none;
        cursor: pointer;
        /*margin-left: 2px;*/
        padding: 5px 10px;
    }
    .tab_box div{
        height: 200px;
        border: solid 1px;
    }
    .selected{
        background-color: orange;
        color: #fff;
    }
    .hover{
        background-color: pink;
    }
</style>

<div class="tab">
    <div class="tab_menu">
        <ul>
            <li  id="tab1">时事</li>
            <li class="selected" id="tab2">新闻</li>
            <li id="tab3">体育</li>
        </ul>
    </div>
    <div class="tab_box">
        <div class="tab_box_tab1">时事</div>
    <div class="tab_box_tab2">新闻</div>
    <div class="tab_box_tab3">体育</div>
    </div>
</div>


<script>
    var selectTab = $(".tab_menu li.selected").attr("id");
    $(".tab_box").find(".tab_box_" + selectTab).show().siblings().hide();
    $(".tab_menu li").hover(function() {
        $(this).addClass("hover");
    }, function() {
        $(this).removeClass("hover");
    }).mousedown(function() {
        $(this).addClass("selected").siblings().removeClass("selected");
        var index = $(".tab_menu li").index(this);
        $(".tab_box>div").eq(index).show().siblings().hide();
    });

    $(".tab_menu").on("mousedown", "li", function(e) {
        $(this).addClass("selected").siblings().removeClass("selected");
        $(".tab_box").find(".tab_box_" + this.id).show().siblings().hide();
    })
</script>
```
