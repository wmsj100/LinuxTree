```javascript
    function getNodeSize(node, prop) {
        //判断节点是不是`DOM`节点
        if (typeof node.ownerDocument !== "object") {
            return null;
        }

        var styleArr = null;

        try {
            styleArr = document.defaultView.getComputedStyle(node, null);
        } catch (ex) {
            styleArr = node.currentStyle;
        }
        return styleArr[prop];
    }
```
